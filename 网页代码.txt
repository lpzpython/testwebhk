        // 渲染Tab内容 + 控制首页图片显示/隐藏
        function renderTabContent(tabKey, sectionKey) {
            const tabData = siteContent[tabKey];
            const sectionData = tabData.sections[sectionKey];
            // 1. 更新主标题
            mainTitle.textContent = sectionData.title;
            // 2. 控制首页图片：仅【关于本站-香港亞文化監察】显示，其他Tab隐藏
            if (tabKey === 'about' && sectionKey === 'focus') {
                homeImageWrap.style.display = 'block';
            } else {
                homeImageWrap.style.display = 'none';
            }
            // 3. 清空内容容器
            contentContainer.innerHTML = '';

            // 渲染不同Tab的内容
            if (sectionKey === 'declaration') {
                // 渲染声明（规整排版）
                const declarationCard = document.createElement('div');
                declarationCard.className = 'content-card declaration-content';
                declarationCard.innerHTML = `
                    <div class="card-header">
                        <span>${sectionData.title}</span>
                        <span>&gt;</span>
                    </div>
                    <div class="card-content">
                        ${sectionData.content.map(item => {
                            if (item.type === 'paragraph') return `<p>${item.text.replace(/\n/g, '<br>')}</p>`;
                            if (item.type === 'heading') return `<h3>${item.text}</h3>`;
                            if (item.type === 'list') return `<ul>${item.items.map(i => `<li>${i.replace(/\n/g, '<br>')}</li>`).join('')}</ul>`;
                        }).join('')}
                    </div>
                `;
                contentContainer.appendChild(declarationCard);
            } else if (sectionKey === 'articles') {
                // 渲染文章（左标题+中封面）
                const articlesCard = document.createElement('div');
                articlesCard.className = 'content-card articles-content';
                articlesCard.innerHTML = `

                        <span>${sectionData.title}</span>
                        <span>&gt;</span>
                    </div>
                    <div class="card-content">
                        <div class="article-grid">
                            ${sectionData.content.map(article => `
                                <div class="article-card">
                                    <div class="article-title">${article.title}</div>
                                    <a href="${article.url}" target="_blank">
                                        <img src="${article.image}" alt="${article.title}" class="article-image">
                                    </a>
                                </div>
                            `).join('')}
                        </div>
                    </div>
                `;
                contentContainer.appendChild(articlesCard);
            } else if (sectionKey === 'videos') {
                // 渲染影音（修复点击播放逻辑）
                const videosCard = document.createElement('div');
                videosCard.className = 'content-card videos-content';

                    <div class="card-header">
                        <span>${sectionData.title}</span>
                        <span>&gt;</span>
                    </div>
                    <div class="card-content">

                            ${sectionData.content.map((video, index) => `
                                <div class="video-card" data-embed="${video.embed}" data-title="${video.title}">

        // 视频点击播放核心函数（修复版）
        function bindVideoClick() {
            const allVideoCards = document.querySelectorAll('.video-card');
            allVideoCards.forEach(card => {
                const thumbnail = card.querySelector('.video-thumbnail');
                thumbnail.addEventListener('click', function() {
                    const embedUrl = card.dataset.embed;
                    const videoTitle = card.dataset.title;
                    // 点击封面后替换为YouTube播放器（适配所有设备）
                    card.innerHTML = `
                        <div class="video-title">${videoTitle}</div>
                        <iframe src="${embedUrl}" title="${videoTitle}" frameborder="0" 
                        allow="accelerometer; autoplay; clipboard-write; encrypted-media; gyroscope; picture-in-picture; web-share" 
                        referrerpolicy="strict-origin-when-cross-origin" allowfullscreen class="video-iframe"></iframe>
                    `;
                });
            });
        }

        // 导航Tab点击事件
        navTabs.forEach(tab => {
            tab.addEventListener('click', (e) => {
                e.stopPropagation();
                const tabKey = tab.dataset.tab;
                const firstSectionKey = Object.keys(siteContent[tabKey].sections)[0];
                renderTabContent(tabKey, firstSectionKey);
            });
        });

        // 子菜单点击事件
        submenuItems.forEach(item => {
            item.addEventListener('click', (e) => {
                e.stopPropagation();
                const tabKey = item.closest('.nav-tab').dataset.tab;
                const sectionKey = item.dataset.section;
                renderTabContent(tabKey, sectionKey);
            });
        });

        // 页面加载完成初始化
        window.addEventListener('DOMContentLoaded', initPage);
